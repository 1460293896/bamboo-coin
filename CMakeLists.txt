 cmake_minimum_required(VERSION 2.8.12)
 project(SimpleCoin)

 
 set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++17")

 include(${CMAKE_BINARY_DIR}/build/conanbuildinfo.cmake)
 conan_basic_setup()

include_directories(/usr/local/include/)

file(GLOB CORE_SOURCES "src/core/*.cpp")
file(GLOB CRYPTO_SOURCES "src/external/ed25519/*.c")

 add_executable(tests ${CORE_SOURCES} ${CRYPTO_SOURCES} ./src/tests/tests.cpp)
 add_executable(server ${CORE_SOURCES} ${CRYPTO_SOURCES} ./src/tools/server.cpp)
 add_executable(cli ${CORE_SOURCES} ${CRYPTO_SOURCES} ./src/tools/cli.cpp)
 add_executable(miner ${CORE_SOURCES} ${CRYPTO_SOURCES} ./src/tools/miner.cpp)
 add_executable(keygen ${CORE_SOURCES} ${CRYPTO_SOURCES} ./src/tools/keygen.cpp)
 add_executable(simulate ${CORE_SOURCES} ${CRYPTO_SOURCES} ./src/tools/simulate.cpp)
 add_executable(benchmark ${CORE_SOURCES} ${CRYPTO_SOURCES} ./src/tools/benchmark.cpp)
 if (APPLE)
    target_link_libraries(tests ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.dylib)
    target_link_libraries(server ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.dylib)
    target_link_libraries(cli ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.dylib)
    target_link_libraries(miner ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.dylib)
    target_link_libraries(keygen ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.dylib)
    target_link_libraries(simulate ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.dylib)
    target_link_libraries(benchmark ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.dylib)
endif()

if (UNIX AND NOT APPLE)
    target_link_libraries(tests ${CONAN_LIBS} /usr/lib/x86_64-linux-gnu/libleveldb.so -lstdc++fs)
    target_link_libraries(server ${CONAN_LIBS} /usr/lib/x86_64-linux-gnu/libleveldb.so -lstdc++fs)
